## 24、两两交换链表中的节点
> 第四天，第一次感觉到一点点轻舟已过万重山的感觉，虽然leetcode官网给这道题目定的难度是**中**，但是这是我为数不多写得得心应手的题目
#### 下面是我美如画的初解
```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* swapPairs(ListNode* head) {
        ListNode* dummyhead = new ListNode(0);
        dummyhead->next = head;
        ListNode* tmp = dummyhead;

        while (tmp->next != NULL && tmp->next->next != NULL) {
            ListNode* x = tmp->next->next;
            tmp->next->next = x->next;
            x->next = tmp->next;
            tmp->next = x;
            tmp = tmp->next->next;
        }

        return dummyhead->next;
    }
};
```
> 要注意的一点就是记得删除创建的虚拟节点dummyhead  
#### 以下是官方题解
```c++
class Solution {
public:
    ListNode* swapPairs(ListNode* head) {
        ListNode* dummyHead = new ListNode(0); // 设置一个虚拟头结点
        dummyHead->next = head; // 将虚拟头结点指向head，这样方便后面做删除操作
        ListNode* cur = dummyHead;
        while(cur->next != nullptr && cur->next->next != nullptr) {
            ListNode* tmp = cur->next; // 记录临时节点
            ListNode* tmp1 = cur->next->next->next; // 记录临时节点

            cur->next = cur->next->next;    // 步骤一
            cur->next->next = tmp;          // 步骤二
            cur->next->next->next = tmp1;   // 步骤三

            cur = cur->next->next; // cur移动两位，准备下一轮交换
        }
        ListNode* result = dummyHead->next;
        delete dummyHead;
        return result;
    }
};
```
## 19.删除链表的倒数第N个元素
> 主要就是一个双指针的应用，代码部分没有问题，感觉可以多考虑一点：N是否越界，但是题目貌似没有考虑到
```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        ListNode* dummyhead = new ListNode(0);
        dummyhead->next = head;
        ListNode* slow = dummyhead;
        ListNode* fast = dummyhead;
        for (int i = 0; i < n - 1; i++) {
            if (fast->next != NULL) {
                fast = fast->next;
            }
            else {
                return head;
            }
        }

        ListNode* tmp = slow;
        while (fast->next != NULL) {
            fast = fast->next;
            tmp = slow;
            slow = slow->next;
        }

        tmp->next = slow->next;
        delete slow;

        // 返回新的链表头部
        ListNode* result = dummyHead->next;
        delete dummyHead; // 清理虚拟头节点
        return result;
    }
};
```
## 面试题02.07.链表相交
在理清思路之后，写代码就感觉水到渠成了，要注意的主要就是两点：
+ 不是哪里都需要使用虚拟指针，比如这道题目就是
+ 如果在某一点相交，那么后面必定都相等，因为在某个点处所指向的地址确定了，指向的地址所指向的地址也是确定的
```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        // 建立两个虚拟节点
        ListNode* dummynode1 = new ListNode(0);
        ListNode* dummynode2 = new ListNode(0);
        dummynode1->next = headA;
        dummynode2->next = headB;

        // 先遍历一遍，查询链表长度
        int i = 0;
        int j = 0;
        while (dummynode1->next != NULL) {
            dummynode1 = dummynode1->next;
            ++i;
        }
        while (dummynode2->next != NULL) {
            dummynode2 = dummynode2->next;
            ++j;
        }

        // 判断最后一个元素是否相同，再决定是否继续
        if (dummynode1 != dummynode2) {
            return NULL;
        }

        dummynode1 = headA;
        dummynode2 = headB;

        if (i >= j) {
            for (int x = 0; x < i - j; ++x) {
                dummynode1 = dummynode1->next;
            }
        }
        if (i < j) {
            for (int x = 0; x < j - i; ++x) {
                dummynode2 = dummynode2->next;
            }
        }

        while (dummynode1 != NULL) {
            if (dummynode1 == dummynode2) {
                return dummynode1;
            }
            dummynode1 = dummynode1->next;
            dummynode2 = dummynode2->next;
        }

        return NULL;
    }
};
```
